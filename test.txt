git remote  // 원격저장소 확인 

git remote -v // 별명과 주소 전제 원격저장소 확인 

git remote rm prcal  // 원격 저장소 삭제 

git push -u prcal master 
// 로컬 저장소에 있는 master 브랜치의 내용이
prcal 이라고 하는 <url> 있는 원격 저장소로 저장 
 -u 로 디폴트로 master 브랜치랑 연동하기 때문에 

그래서 master 브랜치 수정하면 
git commit -am "msg"
